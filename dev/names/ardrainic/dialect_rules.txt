# ============================================================
# SITTADELLIAN (Picardie/Normandie — dur, fortifie)
# ============================================================
# Terrain: forets nord, fortifie
# Shift: consonnes durcies, terminaisons fermees
#
# -val  → -vard
# -ain  → -art
# -en   → -ent
# -or   → -ort
# -el   → -eld
# -in   → -ird
# -elin → -elard
# -ore  → -ort
# -ande → -arde
# -ise  → -ise (inchange)
# -elle → -eld
#
# Exemples:
# Thibeval → Thibevard | Corvain → Corvart | Galenor → Galenort
# Belisore → Belisort  | Alisande → Alisarde

# ============================================================
# AULLERICAN (Artois/Flandre — nord-frontiere, touche germanique)
# ============================================================
# Terrain: plaines nord, frontiere torrent
# Shift: diphtongues germanisantes legeres
#
# -val  → -vael
# -ain  → -aen
# -or   → -oer
# -en   → -aen
# -elin → -elin (inchange — deja doux)
# -ore  → -oer
# -ande → -aende
#
# Exemples:
# Thibeval → Thibevael | Corvain → Corvaen | Galenor → Galenoer
# Belisore → Belisoer  | Alisande → Alisaende

# ============================================================
# ALOISOISE (Lorraine/Champagne — est, influence germanique marquee)
# ============================================================
# Terrain: collines est, contact torrent
# Shift: ajout -en/-er (rallongement lorrain)
#
# -val  → -valen
# -ain  → -ainer
# -or   → -oren
# -en   → -ener
# -elin → -eliner
# -ore  → -oren
# -ande → -anden
#
# Exemples:
# Thibeval → Thibevalen | Corvain → Corvainer | Galenor → Galenoren
# Belisore → Belisoren  | Alisande → Alisanden

# ============================================================
# CAPRAINOIS (Loire/Anjou — doux, liquide, ligerian)

#
# -val  → -vel
# -ain  → -in
# -or   → -orel
# -en   → -en (inchange)
# -el   → -el (inchange)
# -elin → -elin (inchange)
# -ore  → -orel

# Exemples:

# Belisore → Belisorel | Alisande → Alisandel

# ============================================================
# PWASCIARDS (Poitou/Vendee — maritime, tronque)
# ============================================================
# Terrain: cote ouest, marins
# Shift: tronque la derniere syllabe/consonne
#
# -val  → -va
# -ain  → -a
# -or   → -o
# -en   → -e
# -elin → -el
# -ric  → -ri
# -ore  → -o
# -ande → -a
# -ise  → -is
# -elle → -el
#
# Exemples:
# Thibeval → Thibev | Corvain → Corva | Galenor → Galeno
# Belisore → Beliso | Alisande → Alisa | Nimue → Nimu

# ============================================================
# SERENISH (Berry/Auvergne — montagnard, archaique)
# ============================================================

# -en   → -on
# -in   → -on

# -or   → -or (inchange — deja archaique)

# -ande → -ande (inchange — deja archaique)
#
# Exemples:
# Thibeval → Thibevan | Corvain → Corvan | Galenor → Galenor

# ============================================================
# APPAVIN (Alpes — franco-alpin, ex-Torrent)
# ============================================================
# Terrain: montagnes est, isole, influence germanique residuelle
# Ex-culture Torrent, deplacee vers Ardrainic
# Shift: vocalique montagnard a→o, diminutifs -el/-et
#
# -val  → -vel
# -ain  → -on
# -en   → -et
# -or   → -or (inchange — montagnard)
# -elin → -elet
# -ric  → -ret
# -ore  → -oret
# -ande → -ondel
# -ise  → -iset
# -elle → -ellet
#
# Touche germanique residuelle:
# Quelques noms gardent un -k/-g terminal rare (heritage torrent)
# Ex: Aldric → Aldret (pas Aldrik — plus assez germanique)
#
# Exemples:
# Thibeval → Thibevel  | Corvain → Corvon   | Galenor → Galenor
# Belisore → Belisoret | Alisande → Alisondel | Nimue → Nimuet
# Aldric → Aldret      | Florentin → Florentet
